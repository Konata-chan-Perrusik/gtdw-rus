exploitative_tax_collector = {
	levies_factor = -0.9
	tax_factor = 0.5

	liege_modifier = {
		domain_tax_mult = 0.1
		county_opinion_add = -5
	}
	subject_modifier = {
		monthly_county_control_growth_factor = -0.1
		domain_tax_mult = 0.2
		county_opinion_add = -20
		development_growth_factor = -0.05
		opinion_of_liege = -5
	}

	is_valid = {
		custom_description = {
			text = vassal_has_the_wrong_religion
			subject = scope:vassal
			NOT = { scope:vassal.faith ?= scope:liege.faith }
		}
		scope:liege.culture = {
			has_innovation = innovation_bannus
		}
	}

	ai_will_do = {
		value = default_tax_obligation_ai_will_do

		if = {
			limit = {
				scope:liege = {
					ai_greed >= 75 # The AI has to be very greedy
					ai_compassion <= 0 # And not kind enough to have qualms about it
				}
			}
			add = 2
		}
		if = {
			limit = {
				scope:liege = {
					any_vassal = {
						percent >= 0.25
						NOT = { faith = scope:liege.faith }
					}
				}
			}
			add = 5
		}
	}
}
