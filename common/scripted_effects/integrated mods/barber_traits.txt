toggle = {
	if = {
		limit = { has_variable = $var$ }
		remove_variable = $var$
	}
	else = {
		set_variable = $var$
	}
}

#could've probable done this better through flags
double_toggle = {
	if = {
		limit = { has_variable = $var$ }
		remove_variable = $var$
	}
	else = {
		set_variable = $var$
		if = {
			limit = { has_variable = $var2$ }
			remove_variable = $var2$
		}
	}
}

triple_toggle = {
	if = {
		limit = { has_variable = $var$ }
		remove_variable = $var$

	}
	else = {
		set_variable = $var$
		remove_variable = $var2$
		remove_variable = $var3$
	}
}

four_toggle = {
	if = {
		limit = { has_variable = $var$ }
		remove_variable = $var$
	}
	else = {
		set_variable = $var$
		remove_variable = $var2$
		remove_variable = $var3$
		remove_variable = $var4$
	}
}
