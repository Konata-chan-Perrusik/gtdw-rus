set_province_feature = {
	$TITLE$ = {
		if = {
			limit = { tier > tier_county }
			every_in_de_jure_hierarchy = {
				limit = {
					tier = tier_county
					NOT = { is_in_list = excluded_provinces }
				}
				add_county_modifier = $MODIFIER$
			}
		}
		else = { add_county_modifier = $MODIFIER$ }
	}
}

set_arcane_fallout = {
	$TITLE$ = {
		if = {
			limit = { tier > tier_county }
			every_in_de_jure_hierarchy = {
				limit = { tier = tier_county }
				set_variable = {
					name = arcane_fallout
					value = $VALUE$
				}
			}
		}
		else = {
			set_variable = {
				name = arcane_fallout
				value = $VALUE$
			}
		}
	}
}

	if = {
		limit = { 
			has_holding_type = church_holding
		}
		set_holding_type = tribal_holding

	if = {

	}

	random_list = {
		30 = {
			while = {
				count = $MAJOR$

			trigger = { barony = { is_capital_barony = no } }
			while = {
				count = $MINOR$
				generate_building = yes
			}
		}
	}
}

development_growth_period_effect = {
	if = {
		limit = { current_year >= $START$ }
		if = {
			limit = { current_year < $END$ }
			save_scope_value_as = {
				name = year_diff
				value = {
					value = current_year
					subtract = $START$
				}
			}
		}
		else = {
			save_scope_value_as = {
				name = year_diff
				value = {
					value = $END$
					subtract = $START$
				}
			}
		}
		change_development_progress_with_overflow = {
			value = scope:year_diff
			multiply = $GROWTH_RATE$
		}
	}
}

development_growth_change_factor_effect = {
	if = {
		limit = { current_year >= $YEAR$ }
		save_scope_value_as = {
			name = development_diff
			value = {
				value = development_level
				multiply = $FACTOR$
				floor = yes
			}
		}
		change_development_level = scope:development_diff
	}
}

			}
		}
		every_neighboring_county = {
			limit = {
				NOR = {
					has_county_modifier = fogswept_modifier
					has_county_modifier = bordering_fog_modifier
				}
			}
			add_county_modifier = bordering_fog_modifier
		}
	}
}

########################################
# WASTELAND AND RUIN CONSISTENCY CHECK #
########################################

gh_wasteland_and_ruin_check_effect = { # This checks for various errors related to playable characters with wasteland holdings etc
	every_county = {
		if = {
			limit = {
				holder = { is_ruin = no }
				OR = {
					culture = culture:wasteland
					faith = faith:wasteland_faith
					any_county_province = { has_holding_type = wasteland_empty_holding }
				}
			}
			if = {
				limit = { culture = culture:wasteland }
				set_county_culture = holder.culture
				debug_log = debug_log.non_wasteland_holder.wasteland_culture
				debug_log_scopes = no
			}
			if = {
				limit = { faith = faith:wasteland_faith }
				set_county_faith = holder.faith
				debug_log = debug_log.non_wasteland_holder.wasteland_faith
				debug_log_scopes = no
			}
			if = {
				limit = { any_county_province = { has_holding_type = wasteland_empty_holding } }
				every_county_province = {
					limit = { has_holding_type = wasteland_empty_holding }
					set_holding_type = ruin_holding
					province_owner = {
						if = {
							limit = { 
								highest_held_title_tier = tier_barony
							}
							change_government = feudal_government
						}
					}
					debug_log = debug_log.non_wasteland_holder.wasteland_holding
					debug_log_scopes = no
				}
			}
		}
		else_if = {
			limit = {
				holder = { is_ruin = yes }
				NOR = {
					culture = culture:wasteland
					faith = faith:wasteland_faith
					any_county_province = {
						OR = {
							has_holding_type = wasteland_empty_holding
							has_holding_type = ruin_holding
							has_holding_type = ruined_district_holding
						}
					}
				}
			}
			if = {
				limit = { NOT = { culture = culture:wasteland } }
				set_county_culture = holder.culture
				debug_log = debug_log.wasteland_holder.non_wasteland_culture
				debug_log_scopes = no
			}
			if = {
				limit = { NOT = { faith = faith:wasteland_faith } }
				set_county_faith = holder.faith
				debug_log = debug_log.wasteland_holder.non_wasteland_faith
				debug_log_scopes = no
			}
			if = {
				limit = {
					any_county_province = {
						NOR = {
							has_holding_type = wasteland_empty_holding
							has_holding_type = ruin_holding
							has_holding_type = ruined_district_holding
						}
					}
				}
				every_county_province = {
					limit = {
						NOR = {
							has_holding_type = wasteland_empty_holding
							has_holding_type = ruin_holding
							has_holding_type = ruined_district_holding
						}
					}
					set_holding_type = ruin_holding
					debug_log = debug_log.wasteland_holder.non_wasteland_holding
					debug_log_scopes = no
				}
			}
		}
	}
}
