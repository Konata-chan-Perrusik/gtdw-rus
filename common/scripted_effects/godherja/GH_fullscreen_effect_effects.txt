# Scope: Any
# COLOR - name of the overlay color (red or green)
gh_fullscreen_effect_overlay_fade_in_step_effect = {
	if = {
		limit = { NOT = { exists = global_var:gh_fullscreen_$COLOR$_overlay_opacity } }

		set_global_variable = {
			name = gh_fullscreen_$COLOR$_overlay_opacity
			value = 0.0
		}
	}

	change_global_variable = {
		name = gh_fullscreen_$COLOR$_overlay_opacity
		add  = gh_fullscreen_effect_fade_in_step
	}
}

# Scope: Any
# COLOR - name of the overlay color (red or green)
gh_fullscreen_effect_overlay_fade_out_step_effect = {
	change_global_variable = {
		name     = gh_fullscreen_$COLOR$_overlay_opacity
		subtract = gh_fullscreen_effect_fade_out_step
	}

	if = {
		limit = { global_var:gh_fullscreen_$COLOR$_overlay_opacity <= 0.0 }

		remove_global_variable = gh_fullscreen_$COLOR$_overlay_opacity
	}
}
