lower_hunter_xp_effect = {
	if = {
		limit = { has_trait = lifestyle_hunter }
		if = {
			limit = {
				has_trait = lifestyle_hunter
				has_trait_xp = {
					trait = lifestyle_hunter
					track = beasthunter
					value >= 1
				}
			}
			add_trait_xp = {
				trait = lifestyle_hunter
				track = beasthunter
				value = $VALUE$
			}
		}
		if = {
			limit = {
				has_trait = lifestyle_hunter
				has_trait_xp = {
					trait = lifestyle_hunter
					track = falconer
					value >= 1
				}
			}
			add_trait_xp = {
				trait = lifestyle_hunter
				track = falconer
				value = $VALUE$
			}
		}
		else = {
			add_trait_xp = {
				trait = lifestyle_hunter
				track = hunter
				value = $VALUE$
			}
		}
	}
}
