migrate_holy_sites_special_building_slot_effect = {
	if = {
		limit = { religion = { is_in_family = gh_marcher } }
		every_holy_site = {
			limit = { title_province = { has_special_building_slot = no } }
			title_province = { add_special_building_slot = 01_holy_site_marcher_cathedral }
		}
	}
	else_if = {
		limit = {
			religion = {
				is_in_family = rf_eastern
				NOT = { this = religion:fourteen_heavenly_methods_religion }
			}
		}
		every_holy_site = {
			limit = { title_province = { has_special_building_slot = no } }
			title_province = { add_special_building_slot = holy_site_indian_grand_temple_01 }
		}
	}
	else_if = {
		limit = { religion = { is_in_family = rf_pagan } }
		every_holy_site = {
			limit = { title_province = { has_special_building_slot = no } }
			title_province = { add_special_building_slot = holy_site_pagan_grand_temple_01 }
		}
	}
	else_if = {
		limit = { religion = { is_in_family = gh_rf_aversarian } }
		every_holy_site = {
			limit = { title_province = { has_special_building_slot = no } }
			title_province = { add_special_building_slot = 01_holy_site_hallowed_naon }
		}
	}
	else = {
		every_holy_site = {
			limit = { title_province = { has_special_building_slot = no } }
			title_province = { add_special_building_slot = holy_site_other_grand_temple_01 }
		}
	}
}
