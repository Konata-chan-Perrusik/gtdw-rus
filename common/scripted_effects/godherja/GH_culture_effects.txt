chort_traits_more_valued_effect = {
	if = {
		limit = { culture = { has_cultural_parameter = chort_traits_more_valued } }
		add_prestige = { value = medium_prestige_value }
	}
}

gh_steppe_queen_xp_gain_effect = {
	if = {
		limit = { eligible_for_gh_steppe_queenhood = yes }
		if = {
			limit = { NOT = { has_trait = lifestyle_gh_steppe_queen } }
			add_trait = lifestyle_gh_steppe_queen
		}
		add_trait_xp = {
			trait = lifestyle_gh_steppe_queen
			value = $VALUE$
		}
		GH_achiev_rapunzel_check = yes
	}
}
