# LEGON
legon_title_accepted_effect = {
	basic_gained_court_position_effect = yes
	scope:employee = {
		if = {
			limit = {
				NOT = {
					OR = {
						has_trait = gh_legon
						has_trait = gh_arch_legon
					}
				}
			}
			add_trait = gh_legon
			add_character_flag = granted_legon_by_court_position
		}
	}
}
legon_title_revoked_effect = {
	basic_revoked_court_position_effect = yes
	scope:employee = {
		if = {
			limit = {
				has_trait = gh_legon
				has_character_flag = granted_legon_by_court_position
			}
			remove_trait = gh_legon
			remove_character_flag = granted_legon_by_court_position
		}
	}
}
legon_title_invalidated_effect = {
	basic_invalidated_court_position_effect = yes
	scope:employee = {
		if = {
			limit = {
				is_alive = yes
				has_trait = gh_legon
				has_character_flag = granted_legon_by_court_position
			}
			remove_trait = gh_legon
			remove_character_flag = granted_legon_by_court_position
		}
	}
}
