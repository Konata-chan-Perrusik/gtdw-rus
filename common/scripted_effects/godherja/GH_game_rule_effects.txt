game_rule_special_hammarts_wild_ride_effect = {
	every_living_character = {
		if = {
			limit = {
				OR = {
					NOT = { any_held_county = { any_county_province = { geographical_region = world_ga_marcher_kingdoms } } }
					has_title = title:d_callesc
				}
			}
			remove_magical_trait_effect = yes
			#remove_magical_education_effect = yes GH_TODO: update for aspects
			add_trait = magic_good_20
			if = {
			    limit = {
					NOT = { can_access_magic = yes }
			    }
				learn_random_culture_magic_school_effect = {
					MIN_SPELLS = generated_character_min_spells_default_value
					MAX_SPELLS = generated_character_max_spells_default_value
					MAX_SPELL_LEVEL = generated_character_max_spell_level_default_value
				}
			}
		}
	}
}
