	trigger = {
		OR = {
			AND = {
				scope:culture = { has_cultural_pillar = heritage_damotan }
				OR = { 
					scope:other_culture = { has_cultural_pillar = heritage_sarradonian }
					scope:other_culture = { has_cultural_pillar = heritage_dawic }
					scope:other_culture = { has_cultural_pillar = heritage_marmaladid }
					scope:other_culture = { has_cultural_pillar = heritage_dezhunic }
				}
			}
			AND = {
				OR = { 
					scope:culture = { has_cultural_pillar = heritage_sarradonian }
					scope:culture = { has_cultural_pillar = heritage_dawic }
					scope:culture = { has_cultural_pillar = heritage_marmaladid }
					scope:culture = { has_cultural_pillar = heritage_dezhunic }
				}
				scope:other_culture = { has_cultural_pillar = heritage_damotan }

		}
	}
	hybrid = yes
}

		OR = {
			AND = {
				scope:culture = { has_cultural_pillar = heritage_nattanan }
				scope:culture = { has_cultural_tradition = tradition_kedari_marked }
				scope:other_culture = { has_cultural_pillar = heritage_easterner }
				scope:other_culture = { has_cultural_pillar = ethos_bureaucratic }
			}
			AND = {
				scope:culture = { has_cultural_pillar = heritage_easterner }
				scope:culture = { has_cultural_pillar = ethos_bureaucratic }
				scope:other_culture = { has_cultural_pillar = heritage_nattanan }
				scope:other_culture = { has_cultural_tradition = tradition_kedari_marked }
			}
		}
	}
	hybrid = yes
}
