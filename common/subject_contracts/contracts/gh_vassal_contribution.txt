pirate_government_obligations = {
	uses_opinion_of_liege = yes
	obligation_levels = {
		default = { # Pirates
			levies = {
				value = 0
				max = 0.5
				scope:liege = {
					add = {
						value = 0.01
						multiply = dread
					}
				}
				if = {
					limit = { exists = scope:tax_collector }
					add = 0.02
					multiply = pirate_government_tax_collector_aptitude_multiplier
				}
			}
			tax = {
				value = 0
				max = 0.5
				scope:liege = {
					add = {
						value = 0.01
						multiply = dread
					}
				}
				if = {
					limit = { exists = scope:tax_collector }
					add = 0.02
					multiply = pirate_government_tax_collector_aptitude_multiplier
				}
			}

			ai_liege_desire = @ai_standard_liege_desire
			ai_subject_desire = 0
		}

	}
}
