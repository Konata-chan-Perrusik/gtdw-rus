# Used for standard low-ish skill contributions across most schemes; if you need a different value, you can just use a non-scripted modifier.
# This will add whichever skill you input to Arcana and then divide both
scheme_type_skill_and_arcana_success_chance_modifier = {
	modifier = {
		add = {
			value = 0
			add = scope:owner.$SKILL$
			add = scope:owner.character_arcana_baseline_value
			multiply = 0.5
		}
		desc = SCHEME_YOUR_$SKILL$_AND_ARCANA
	}
}
