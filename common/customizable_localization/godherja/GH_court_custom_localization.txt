# Court Magi
# GetActualCourtMagiPossessive should be updated synchronously with this
GetActualCourtMagiTitle = {
	type = character

	#Fogeater
	text = {
		trigger = { is_fogeater = yes }
		localization_key = councillor_court_magi_fogeater
	}

	#Redlander
	text = {
		trigger = { is_redlander = yes }
		localization_key = councillor_court_magi_redlander
	}

	#Tribal
	text = {
		trigger = {
			GH_is_any_tribal_government_trigger = yes
			liege.highest_held_title_tier >= tier_empire
		}
		localization_key = councillor_court_magi_tribal_empire
	}
	text = {
		trigger = {
			GH_is_any_tribal_government_trigger = yes
			liege.highest_held_title_tier >= tier_kingdom
		}
		localization_key = councillor_court_magi_tribal_kingdom
	}
	text = {
		trigger = {
			GH_is_any_tribal_government_trigger = yes
			liege.highest_held_title_tier >= tier_duchy
		}
		localization_key = councillor_court_magi_tribal_duchy
	}
	text = {
		trigger = {
			GH_is_any_tribal_government_trigger = yes
			liege.highest_held_title_tier < tier_duchy
			cp:councillor_court_magi = {
				is_female = yes
			}
		}
		localization_key = councillor_court_magi_tribal_county_female
	}
	text = {
		trigger = {
			GH_is_any_tribal_government_trigger = yes
			liege.highest_held_title_tier < tier_duchy
			cp:councillor_court_magi = {
				is_female = no
			}
		}
		localization_key = councillor_court_magi_tribal_county_male
	}

	#Fallback
	text = {
		fallback = yes
		localization_key = councillor_court_magi_fallback
	}
}
