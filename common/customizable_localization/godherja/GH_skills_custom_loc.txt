arcana_skill_level = {
	type = character

	text = {
		localization_key = skill_level_terrible
		trigger = { character_arcana_baseline_value < terrible_skill_level }
	}

	text = {
		localization_key = skill_level_poor
		trigger = {
			character_arcana_baseline_value >= terrible_skill_level
			character_arcana_baseline_value < average_skill_level
		}
	}

	text = {
		localization_key = skill_level_average
		trigger = {
			character_arcana_baseline_value >= average_skill_level
			character_arcana_baseline_value < good_skill_level
		}
	}

	text = {
		localization_key = skill_level_good
		trigger = {
			character_arcana_baseline_value >= good_skill_level
			character_arcana_baseline_value < excellent_skill_level
		}
	}

	text = {
		localization_key = skill_level_excellent
		trigger = { character_arcana_baseline_value >= excellent_skill_level }
	}
}
