CooltipQuartermasterSkill = {
	type = character

	text = {
		trigger = {
			NOT = { exists = cp:councillor_quartermaster }
		}
		localization_key = councillor_quartermaster_does_not_exist_desc
		fallback = yes
	}
	text = {
		trigger = { 
			cp:councillor_quartermaster = { stewardship <= mediocre_skill_rating } 
		}
		localization_key = councillor_quartermaster_skill_mediocre_desc
	}
	text = {
		trigger = { 
			cp:councillor_quartermaster = { 
				stewardship > mediocre_skill_rating 
				stewardship <= decent_skill_rating
			} 
		}
		localization_key = councillor_quartermaster_skill_decent_desc
	}
	text = {
		trigger = { 
			cp:councillor_quartermaster = { 
				stewardship > decent_skill_rating 
				stewardship <= very_high_skill_rating
			} 
		}
		localization_key = councillor_quartermaster_skill_high_desc
	}
	text = {
		trigger = { 
			cp:councillor_quartermaster = { 
				stewardship > very_high_skill_rating
			} 
		}
		localization_key = councillor_quartermaster_skill_excellent_desc
	}
}
