county_culture_uses_caste_system = {
	scope = landed_title
	is_shown = {
		culture = { has_cultural_pillar = heritage_damotan }
	}
	is_valid = {
		OR = {
			holder = {
				culture = { has_cultural_parameter = unlocks_lodge_mechanics }
			}
			holder.top_liege = {
				culture = { has_cultural_parameter = unlocks_lodge_mechanics }
			}
		}
	}
}
