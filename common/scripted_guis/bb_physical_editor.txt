  is_shown = {
    exists = var:inherit_body_bb
  }
  effect = {
    if = {
      limit = { exists = var:inherit_body_bb }
      remove_variable = inherit_body_bb
    }
    else = {
      set_variable = inherit_body_bb
    }
  }
}

bb_reset_gene_bb_gene_chin_forward = {
  effect = {
    remove_variable = bb_gene_chin_forward
  }
}

bb_decrease_gene_bb_gene_chin_forward = {
  effect = {
    if = {
      limit = { NOT = { has_variable = bb_gene_chin_forward } }
      set_variable = { name = bb_gene_chin_forward value = 0 }
    }
    change_variable = { name = bb_gene_chin_forward subtract = 0.01 }
		if = { limit = { var:bb_gene_chin_forward = 0 } remove_variable = bb_gene_chin_forward }
  }
  is_valid = { trigger_if = { limit = { has_variable = bb_gene_chin_forward } var:bb_gene_chin_forward > -1 }  }
}
