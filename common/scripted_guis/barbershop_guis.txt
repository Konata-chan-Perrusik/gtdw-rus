undress_bb = {
	scope = character

	is_shown = {
		has_character_flag = is_naked
	}

	is_valid = {
		is_adult = yes
	}

	effect = {
		if = {
			limit = {
				NOT = { has_character_flag = is_naked }
			}
			add_character_flag = is_naked
		}
		else = {
			remove_character_flag = is_naked
		}
	}
}

	scope = character
	effect = {
		if = {
			limit = {
				NOT = { exists = global_var:FSB_is_loaded }
			}
			set_global_variable = {
				name = FSB_is_loaded
				value = yes
			}
		}
	}
}

set_default_zoom_bb = {
	effect = {
		if = {
			limit = {
				NOT = { exists = var:zoom }
			}
			set_variable = {
				name = zoom
				value = 3
			}
		}
	}
}

zoom_in = {
	scope = character

	is_valid = {
		OR = {
			NOT = {
				has_variable = zoom
			}
			AND = {
				has_variable = zoom
				var:zoom < 5
			}
		}
	}

	effect = {
		if = {
			limit = {
				NOT = { has_variable = zoom }
			}
			set_variable = {
				name = zoom
				value = 3
			}
		}
		change_variable = {
			name = zoom
			add = 1
		}
	}
}

zoom_out = {
	scope = character

	is_valid = {
		OR = {
			NOT = {
				has_variable = zoom
			}
			AND = {
				has_variable = zoom
				var:zoom > 1
			}
		}
	}

	effect = {
		if = {
			limit = {
				NOT = { has_variable = zoom }
			}
			set_variable = {
				name = zoom
				value = 3
			}
		}
		change_variable = {
			name = zoom
			add = -1
		}
	}
}

restore_zoom = {
	is_shown = {
		has_variable = zoom
	}
}

clear_zoom = {
	effect = {
		remove_variable = zoom
	}
}

bigger_text_size_barber = {
    scope = character

    is_valid = {
        trigger_if = {
            limit = { exists = var:barber_text_size }
            var:barber_text_size < 5
        }
    }

    effect = {
        if = {
            limit = {
                NOT = { has_variable = barber_text_size }
            }
            set_variable = {
                name = barber_text_size
                value = 1
            }
        }
        change_variable = {
            name = barber_text_size
            add = 0.1
        }
    }
}

add_children_to_barbershop = {
	saved_scopes = { player }
	effect = {
		clear_variable_list = barber_children
		every_child = {
			limit = {
				is_alive = yes
				NOT = {
					is_primary_heir_of = root
				}
			}
			root = {
				add_to_variable_list = {
					name = barber_children
					target = prev
				}
			}
			scope:player = {
				add_to_variable_list = {
					name = barber_pinned
					target = prev
				}
			}
		}
	}
	is_shown = {
		num_of_living_children > 1
	}
}

bb_copy_dna = {
	effect = {
		if = {
			limit = {
				NOT = { has_variable = bb_original_dna }
			}
			create_character = {
				gender = root
				age = root.age
				culture = root.culture
				faith = root.faith
				location = root.location
				after_creation = {
					copy_inheritable_appearance_from = root
					root = {
						set_variable = {
							name = bb_original_dna
							value = prev
						}
					}
					death = natural
				}
			}
		}
		copy_inheritable_appearance_from = scope:player.var:bb_dna_target
	}
}

	}
