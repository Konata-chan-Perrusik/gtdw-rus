merit_display_with_siblings = {
	scope = character
	
	is_shown = {
		any_parent = {
			has_title = title:e_chevalie
		}
	}
}

merit_display_without_siblings = {
	scope = character
	
	is_shown = {
		AND = {
			culture = { has_cultural_pillar = heritage_marcher }
			NOT = {
				any_parent = {
					has_title = title:e_chevalie
				}
			}
		}
	}
}

legitimacy_display = {
	scope = character
	
	is_shown = {
		OR = {
			GH_is_any_republic_government_trigger = yes 
			GH_is_any_republic_government_trigger = yes
		}
	}
}

wasteland_check = {
	scope = character
	
	is_shown = {
		is_ruin = no
	}
}
