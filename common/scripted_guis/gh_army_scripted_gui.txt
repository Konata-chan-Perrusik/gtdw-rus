raiding_redistribution_scripted_gui = {
	scope = army
	is_shown = {
		army_owner.culture = { has_cultural_parameter = can_distribute_raid_loot_for_prestige_and_development }
	}
	
	effect = {
		if = {
			limit = { exists = var:raiding_redistribution_disabled }
			remove_variable = raiding_redistribution_disabled
		}
		else = {
			set_variable = raiding_redistribution_disabled
		}
	}
}

gh_culture_has_redistribution_scripted_gui = {
	scope = culture
	is_valid = {
		has_cultural_parameter = can_distribute_raid_loot_for_prestige_and_development
	}
}
