arcana_cognition_perk1_tree2 = { # Extra gold from selling Magic Artifacts
	lifestyle = arcana_lifestyle
	tree = arcana_tree2
	position = { 0 0 }
	icon = node_martial

	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_trait = education_arcana
			}
			add = 1989
		}
		if = {
			limit = {
				has_focus = arcana_cognition_focus
			}
			multiply = 5
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}
			multiply = 0
		}
		if = {
			limit = {
				faith = { has_doctrine_parameter = magic_illegal }
			}
			multiply = 0
		}
	}

	can_be_picked = {
		always = yes
	}

	effect = {
		custom_description_no_bullet = {
			text = arcana_cognition_extra_gold_from_magic_artifacts
		}
	}
}
