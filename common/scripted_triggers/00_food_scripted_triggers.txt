# VEGETABLE TRIGGERS

province_has_vegetable_carrot_trigger = {
	OR = {
		geographical_region = GH_REPLACE_world_india
		geographical_region = GH_REPLACE_world_middle_east
		geographical_region = GH_REPLACE_world_europe
		geographical_region = GH_REPLACE_world_steppe_west
		geographical_region = GH_REPLACE_world_asia_minor
	}
}

province_has_vegetable_parsnip_trigger = {
	NOT = { geographical_region = GH_REPLACE_world_africa }
}

province_has_vegetable_turnip_trigger = {
	NOT = { geographical_region = GH_REPLACE_world_africa }
}

province_has_vegetable_yams_trigger = {
	OR = {
		# African yams
		geographical_region = GH_REPLACE_world_africa
		# Chinese yams
		geographical_region = GH_REPLACE_world_tibet
		geographical_region = GH_REPLACE_world_burma
		geographical_region = GH_REPLACE_world_steppe_east
		geographical_region = GH_REPLACE_world_steppe_tarim
	}
}

province_has_vegetable_kokoro_trigger = {
	geographical_region = GH_REPLACE_world_africa
}

province_has_vegetable_ginger_trigger = {
	OR = {
		geographical_region = GH_REPLACE_world_india
		geographical_region = GH_REPLACE_world_burma
	}
}

province_has_vegetable_taro_trigger = {
	OR = {
		geographical_region = GH_REPLACE_world_india
		geographical_region = GH_REPLACE_world_burma
		geographical_region = GH_REPLACE_world_tibet
		geographical_region = GH_REPLACE_world_steppe_east
		geographical_region = GH_REPLACE_world_steppe_tarim
	}
}
