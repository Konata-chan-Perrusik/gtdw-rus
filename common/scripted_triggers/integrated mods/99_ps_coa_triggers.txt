coa_switch_to_mena_trigger = {
	holder = {
		OR = {			
			culture = { has_coa_gfx = arabic_group_coa_gfx }
			culture = { has_coa_gfx = berber_group_coa_gfx }
		}
	}
}

ps_coa_western_heraldry_trigger = {
	scope:culture = {
		OR = {
			has_coa_gfx = english_coa_gfx
			has_coa_gfx = frankish_group_coa_gfx
			has_coa_gfx = latin_group_coa_gfx
			has_coa_gfx = iberian_group_coa_gfx
			has_coa_gfx = breton_coa_gfx
			has_coa_gfx = welsh_coa_gfx
			has_coa_gfx = irish_coa_gfx
			has_coa_gfx = scottish_coa_gfx
			has_coa_gfx = german_group_coa_gfx
		}
	}
}
