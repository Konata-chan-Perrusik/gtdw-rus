
##TRIGGER LIST
# has_conventionally_attractive_trigger
# has_conventionally_ugly_trigger
# has_permanent_physical_injury - checks if you have some permanent injury
#

has_conventionally_attractive_trigger = { # GH ADDITION: Added our traits
	OR = {
		has_trait = beauty_good
		AND = {
			has_trait = strong
			NOT = { has_trait = beauty_bad }
		}
	}
}

has_conventionally_ugly_trigger = { # GH ADDITION: Added our traits
	OR = {
		has_trait = beauty_bad
		has_trait = hunchbacked
		has_trait = one_legged
		has_trait = scarred
		has_trait = one_eyed
		has_trait = scaly
		has_trait = great_pox
		has_trait = dwarf
		has_trait = albino
		
		# GH ADDITION: Added our traits
		
		has_trait = gh_necronic_skin
		has_trait = gh_deep_mutation
		has_trait = gh_dehanded
		has_trait = gh_ritual_scars_3
		has_trait = gh_one_handed
	}
}

has_permanent_physical_injury = {
	OR = {
		has_trait = one_legged
		has_trait = one_eyed
		has_trait = scarred
		has_trait = maimed
		is_eunuch_trigger = yes
		has_trait = blind
	}
}
