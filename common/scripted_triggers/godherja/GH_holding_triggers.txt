	# Using the no_wall holding graphics for cities with special walls so that they don't have double walls
	has_building_or_higher = "01_thetowerwatch"
}

	}
}

GH_any_wasteland_holding_trigger = {
	OR = {
		GH_wasteland_empty_holding_trigger = yes
		GH_ruin_holding_trigger = yes
		GH_fogswept_holding_trigger = yes
	}
}

GH_fogswept_holding_trigger = {
	OR = {
		GH_wasteland_empty_holding_trigger = yes
		GH_ruin_holding_trigger = yes
	}
	trigger_if = {
		limit = { exists = county }
		county = { has_county_modifier = fogswept_modifier }
	}
}

GH_fogswept_barony_trigger = {

		GH_ruin_holding_trigger = yes
	}
	trigger_if = {

	}
}
