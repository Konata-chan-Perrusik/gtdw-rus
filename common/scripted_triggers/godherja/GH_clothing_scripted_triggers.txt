should_wear_maxlatl = { 
	# Maxtlatls seem to have been wildly popular historical, as warrior wear, and particularly for the prestigious sort.
	# But armour had function too and we can't have the entire male culture run around in loincloths.
	# So I'm selecting mostly on actual fitness. Open to suggestions!
	NOT = {
		has_trait = humble
	}
	OR = { 
		AND = { #Young warrior
			age < 25
			current_weight > -40
			current_weight < 40
		}
		AND = { #Good warrior
			age < 65
			prowess > 10
			current_weight > -40
			current_weight < 40
		}
		AND = { #Not entirely fit but vain
			has_trait = arrogant
			calc_true_if = {
				amount = 2
				age < 65
				prowess > 10
				current_weight > -40
				current_weight < 40
			}
		}
	}
}
