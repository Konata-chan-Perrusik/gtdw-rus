
has_any_hills_terrain_danger_type = {
	location ?= {
		OR = {
			travel_danger_type = {
				travel_plan = $TRAVEL$
				terrain = hills
			}
			travel_danger_type = {
				travel_plan = $TRAVEL$
				terrain = marcher_hills
			}
		}
	}
}
