# Pirate
province_has_underground_terrain = {
	GH_any_underground_terrain_trigger = yes
	trigger_if = {
		limit = { exists = empire }
		OR = {
			empire = title:e_abyssal_pits
			empire = title:e_western_undercity
			empire = title:e_eastern_undercity
			empire = title:e_red_chasm
			empire = title:e_sunken_depths
		}
	}
	trigger_if = {
		limit = { exists = kingdom }
		kingdom = title:k_the_mistake
	}
}
