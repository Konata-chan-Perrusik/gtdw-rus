	has_clique_government_trigger = yes

crimes_allow_any_punishment_trigger = {
	legal_tradition_has_flag_trigger = { FLAG = crimes_allow_any_punishment }
}

has_any_crime_trigger = {
	OR = {
		has_revoke_title_reason = scope:$WHO$
		has_execute_reason = scope:$WHO$
		has_imprisonment_reason = scope:$WHO$
		has_banish_reason = scope:$WHO$
	}
}


# Tax Collector Triggers

check_bailate_aptitude = {
	$CHARACTER$ = {
		tax_collector_aptitude = {
			tax_slot_type = republic_playable_tax_slot
			value = $VALUE$
		}
	}
}

check_magistrate_aptitude = {

			tax_slot_type = landed_order_tax_slot

		}
	}
}
