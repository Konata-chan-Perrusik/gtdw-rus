# This file contains scripted triggers determining if
# certain portrait effects must be used for the scoped character.

###########################
## Flat Portrait Effects ##
###########################

portrait_effect_black_shade_trigger = {
	is_alive = yes
	has_variable = gh_is_black_shade
}

portrait_effect_vindeon_shade_trigger = {
	OR = {
		AND = {
			is_alive = yes
			has_variable = gh_is_vindeon_shade
		}
		this = character:nathanalmic_nattanan_5
		this = character:nathanalmic_nattanan_6
		culture = culture:fey_culture
	}
	
	
}

## Textured Statue Portrait Effects ##

portrait_effect_ice_textured_statue_trigger = {
	trigger_if = {
		limit = {
			is_alive = yes
		}

		has_variable = gh_is_ice_textured_statue
	}
	trigger_else = {
		death_reason = death_freeze
	}
}
