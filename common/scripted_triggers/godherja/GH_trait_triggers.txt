GH_is_truly_blind_trigger = { # For references to blind trait
	OR = {
		has_trait = blind
		has_trait = gh_nearly_blind
	}
}

has_any_scarred_trait_trigger = { # For any vanilla references to scars
	OR = {
		has_trait = scarred
		has_trait = gh_ritual_scars_1
		has_trait = gh_ritual_scars_2
		has_trait = gh_ritual_scars_3
	}
}
