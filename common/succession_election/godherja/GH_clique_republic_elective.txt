		# doesn't seem necessary to change this from the feudal one either
		
		##########################	Elector voting patterns (cliques)	##########################
		elector_voting_pattern_circumstances_clique_mechanics_modifier = yes
	}
	
	#scope:candidate = Candidate, scope:title = Title, scope:holder_candidate = Candidate currently selected by ruler.
	candidates = {
		add = {
			type = holder_direct_vassals
			limit = {
				GH_is_any_republic_government_trigger = yes
				age > 15
			}
		}
		add = {
			type = title_claimants
			limit = {
				OR = {
					GH_is_any_republic_government_trigger = yes
					is_playable_character = no

				age > 15
			}
		}
		add = {
			type = holder_close_family
			limit = {
				OR = {
					GH_is_any_republic_government_trigger = yes
					is_playable_character = no
				}
				age > 15
				culture = { has_same_culture_heritage = scope:holder.culture }
			}
		}
	}
