### Hunt Intents

slay_beast_intent = {
	icon = acquire_pelts_intent

	is_shown = {
		has_dlc_feature = advanced_activities
		is_alive = yes
		is_adult = yes
		NOT = {
			scope:special_option ?= flag:hunt_type_falconry
			scope:special_option ?= flag:hunt_type_nerge
			NOR = {
				faith = { has_doctrine = tenet_sjalvolki_hunt }
				culture = { has_cultural_tradition = tradition_sacred_hunts }
				culture = { has_cultural_tradition = tradition_hunters }
				has_perk = beasthunter_perk
			}
		}
	}

	is_valid = {
		OR = {
			can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = this }
			faith = { has_doctrine = tenet_sjalvolki_hunt }
			culture = { has_cultural_tradition = tradition_sacred_hunts }
			culture = { has_cultural_tradition = tradition_hunters }
			has_perk = beasthunter_perk
		}
	}

	ai_will_do = {
		value = 50
		add = {
			value = ai_boldness
			divide = 2
		}
		add = {
			value = ai_compassion
			multiply = -1
			divide = 2
		}
		if = {
			limit = { has_trait = lifestyle_hunter }
			add = 50
		}
		if = {
			limit = { has_trait = brave }
			add = 25
		}
		if = {
			limit = { has_trait = craven }
			add = -25
		}
		if = {
			limit = { has_trait = greedy }
			add = 25
		}
		if = {
			limit = { has_trait = generous }
			add = -25
		}
		if = {
			limit = { has_trait = ambitious }
			add = 25
		}
		if = {
			limit = {
				culture = { has_cultural_parameter = cannot_hunt }
			}
			add = -1000
		}
	}

	scripted_animation = {
		animation = hunting_knife_start
	}
}
