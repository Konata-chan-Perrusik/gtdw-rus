# Unique Dynasty Legacies

cenware_legacy_track = { # Blood of the Witch-Breaker
	is_shown = {
		NOT = { has_dlc_feature = legends }
		dynasty = {
			OR = {
				this = dynasty:dyn_sjalvolki_1
				has_dynasty_perk = cenware_legacy_1
			}
		}
	}
}
