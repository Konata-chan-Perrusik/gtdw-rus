				value = agent_trait_bonus_t1_value
				desc = scheme_agent_aptitude.trait.shrewd
			}
		}
		### + Magi
		if = {
			limit = { can_access_magic = yes }
			add = {
				value = agent_trait_bonus_t2_value
				desc = scheme_agent_aptitude.trait.magi
