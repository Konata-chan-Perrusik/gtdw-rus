# goblet_characteristic
	
	goblet_characteristic_large = {
		group = goblet_characteristic
		trigger = {
			artifact_value_trigger = { VALUE = scope:wealth MIN = 10 MAX = 50 }
		}
		weight = {
			value = 4
			# Opakhasians like making big, gaudy monstrosities
			if = {
				limit = {
					trigger_if = {
						limit = { exists = scope:owner.var:artifact_origin_location }
						scope:owner.var:artifact_origin_location = { geographical_region = opakhasia_biozone }
					}
					trigger_else = { always = no }
				}
				add = 2
			}
			# Saw periods of popularity in the late 4th-early 6th centuries and most of the 12th century
			if = {
				limit = {
					trigger_if = {
						limit = { exists = scope:owner.var:goblet_age }
						OR = {
							AND = {
								scope:owner.var:goblet_age >= 351
								scope:owner.var:goblet_age <= 550
							}
							AND = {
								scope:owner.var:goblet_age >= 1090
								scope:owner.var:goblet_age <= 1200
							}
						}
					}
					trigger_else = { always = no }
				}
				add = 2
			}
		}
	}
