	icon = "artifact_statue.dds"
	default_type = sculpture

	#Save the scope value victim_fm to determine if female or male
	#When victim_fm is not saved the statue will pick randomly between female or male

	asset = {
		trigger = {
			OR = {
				AND = {
					exists = scope:victim_fm
					scope:victim_fm = flag:male
				}
				NOT = { exists = scope:victim_fm }
			}
		}
		reference = gh_ice_statue_male_entity
	}
	asset = {
		trigger = {
			OR = {
				AND = {
					exists = scope:victim_fm

				}
				NOT = { exists = scope:victim_fm }
			}
		}
		reference = gh_ice_statue_female_entity
	}
