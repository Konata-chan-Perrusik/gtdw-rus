# % from 0-100 inclusive

outbreak_chance_minor_mult_value = {
	value = 1

	# Game rules
	if = {
		limit = {
			has_game_rule = epidemic_frequency_often
		}
		add = outbreak_increase_mult_low
	}
	else_if = {
		limit = {
			has_game_rule = epidemic_frequency_very_often
		}
		add = outbreak_increase_mult_medium
	}
	else_if = {
		limit = {
			has_game_rule = epidemic_frequency_doomsday
		}
		add = outbreak_increase_mult_high
	}
	else_if = {
		limit = {
			has_game_rule = epidemic_frequency_rare
		}
		add = outbreak_decrease_mult_medium
	}
	else_if = {
		limit = {
			has_game_rule = epidemic_frequency_very_rare
		}
		add = outbreak_decrease_mult_high
	}

	#Development
	add = {
		value = county.development_level
		multiply = 0.002
	}

	# County Control
	if = {
		limit = {
			county = {
				county_control >= 90
				}
			}
		add = outbreak_decrease_mult_low
	}
	else_if = {
		limit = {
			county = {
				county_control <= 25
				}
			}
		add = outbreak_increase_mult_low
	}

	# Epidemic Resistance
	if = {
		limit = {
			county = {
				any_county_province = { epidemic_resistance >= 89 }
				}
			}
		multiply = 0.3
	}
	else_if = {
		limit = {
			county = {
				any_county_province = { epidemic_resistance >= 79 }
				}
			}
		multiply = 0.5
	}
	else_if = {
		limit = {
			county = {
				any_county_province = { epidemic_resistance >= 59 }
				}
			}
		multiply = 0.65
	}
	else_if = {
		limit = {
			county = {
				any_county_province = { epidemic_resistance >= 44 }
				}
			}
		multiply = 0.7
	}
	else_if = {
		limit = {
			county = {
				any_county_province = { epidemic_resistance >= 19 }
				}
			}
		multiply = 0.8
	}
	else_if = {
		limit = {
			county = {
				any_county_province = { 
				NOT = { epidemic_resistance >= 12 }
				}
			}
		}
		multiply = 1.16
	}

	# Liege traits
	if = {
		limit = {
			county = {
				holder = { has_trait = scholar }
				}
			}
		add = outbreak_decrease_mult_low
	}
	if = {
		limit = {
			county = {
				holder = { has_trait = shrewd }
				}
			}
		add = outbreak_decrease_mult_low
	}
	if = {
		limit = {
			county = {
				holder = { has_trait = lifestyle_physician }
				}
			}
		add = outbreak_decrease_mult_medium
	}
	if = {
		limit = {
			county = {
				holder = { has_trait = whole_of_body }
				}
			}
		add = outbreak_decrease_mult_low
	}
	if = {
		limit = {
			county = {
				holder = { has_trait = intellect_good_2 }
				}
			}
		add = outbreak_decrease_mult_low
	}
	if = {
		limit = {
			county = {
				holder = { has_trait = intellect_good_3 }
				}
			}
		add = outbreak_decrease_mult_low
	}
	if = {
		limit = {
			county = {
				holder = { has_trait = lazy }
				}
			}
		add = outbreak_increase_mult_low
	}
	if = {
		limit = {
			county = {
				holder = { has_trait = greedy }
				}
			}
		add = outbreak_increase_mult_low
	}
	if = {
		limit = {
			county = {
				holder = { has_trait = intellect_bad_2 }
				}
			}
		add = outbreak_increase_mult_low
	}
	if = {
		limit = {
			county = {
				holder = { has_trait = intellect_bad_3 }
				}
			}
		add = outbreak_increase_mult_low
	}

	#Era pacing
	if = {
		limit = {
			culture = {
				has_cultural_era_or_later = culture_era_late_medieval
			}
		}
		add = outbreak_decrease_mult_low
	}
	else_if = {
		limit = {
			culture = {
				has_cultural_era_or_later = culture_era_high_medieval
			}
		}
		add = 0
	}
	else_if = {
		limit = {
			culture = {
				has_cultural_era_or_later = culture_era_early_medieval
			}
		}
		add = 0
	}
	else_if = {
		limit = {
			culture = {
				has_cultural_era_or_later = culture_era_tribal
			}
		}
		add = outbreak_decrease_mult_low
	}

	#Terrain
	if = {
		limit = {
			OR = {
				terrain = hills
				terrain = forest
				terrain = desert
				terrain = desert_mountains
				terrain = mountains
				terrain = archipelagic
			}
		}
		add = outbreak_decrease_mult_low
	}
	else_if = {
		limit = {
			OR = {
				terrain = jungle
				terrain = taiga
			}
		}
		add = outbreak_increase_mult_low
	}

	#Developed places
	if = {
		limit = {
			has_holding = yes
		}
		add = outbreak_increase_mult_low
	}
	if = {
		limit = {
			has_holding_type = city_holding
		}
		add = outbreak_increase_mult_low
	}
	if = {
		limit = {
			OR = {
				has_holding_type = herder_holding
				has_holding_type = nomad_holding
			}
		}
		add = outbreak_decrease_mult_high
	}
	
	multiply = 0.8
	min = 0.001
	if = { # Wastelands and Necropoli arent great places for epidemics to spread
		limit = {
			OR = {
				has_holding_type = wasteland_empty_holding
				has_holding_type = necropolis_holding
			}
		}
		multiply = 0
	}
}

			is_target_in_variable_list = {
				name = epidemic_cooldown #Average lifespan
				target = scope:epidemic_type
			}
		}
		if = {
			limit = {
				has_game_rule = epidemic_frequency_very_often
			}
			value = outbreak_decrease_mult_high
		}
		else_if = {
			limit = {
				has_game_rule = epidemic_frequency_doomsday
			}
			value = outbreak_decrease_mult_medium
		}
		else = {
			value = outbreak_decrease_mult_unlikely
		}
		
	}

	#People still remember how to social distance in general
	else_if = {
		limit = {
			has_variable = epidemic_cooldown_general #Average social memory
		}
		if = {
			limit = {
				has_game_rule = epidemic_frequency_very_often
			}
			value = outbreak_decrease_mult_medium
		}
		else_if = {
			limit = {
				has_game_rule = epidemic_frequency_doomsday
			}
			value = outbreak_decrease_mult_low
		}
		else = {
			value = outbreak_decrease_mult_high
		}
	}
}

#How badly effected is a county's development?
epidemic_development_loss_value = {
	subtract = scope:base_loss

	#This feeds into the change_development_progress in 00_epidemics
	#Bear in mind that minimal_development_progress_loss is 10
	#and massive_development_progress_loss is 100
	#Aim for something between those two extremes generally (in minus)

	#Since we have widespread development loss handled by modifier, we
	#now use this figure to 'chunk' down places with very high development
	#so they are disproportionately affected

	scope:epidemic = {
		switch = {
			trigger = outbreak_intensity
			minor = { subtract = 2.5 }
			major = { subtract = 5 }
			apocalyptic = { subtract = 7.5 }
		}
	}

	# Development
	subtract = {
		value = development_level
		multiply = 0.3
	}

	multiply = {
		value = 6
		divide = num_county_holdings
	}

	multiply = {
		value = 100
		subtract = title_province.epidemic_resistance
		multiply = 0.005
		add = 1
	}
}
