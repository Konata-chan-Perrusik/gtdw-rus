		add = 1
	}
	if = {
		limit = { title:e_malcois = { any_in_de_jure_hierarchy = { tier = tier_county } } }
		add = 1
	}
	if = {
		limit = { title:e_aironoi = { any_in_de_jure_hierarchy = { tier = tier_county } } }
		add = 1
	}
	if = {

		add = 1
	}
	min = 2
}
