# STEWARD

steward_develop_county_judicial_council_bonus_monthly_increase = {
	value = 0
	if = {
		limit = {
			exists = liege.dynasty
			liege.dynasty = { has_dynasty_perk = wardenite_legacy_3 }
			has_trait = lifestyle_jurist
		}
		add = steward_develop_county_base_total
		multiply = wardenite_legacy_3_bonus_value
	}
}
