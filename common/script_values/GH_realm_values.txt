faith_hostility_level_conversion_malus = {
	value = 1
	if = {
		limit = {
			faith = {
				faith_hostility_level = {
					target = scope:councillor.faith
					value >= faith_hostile_level
				}
			}
		}
		add = 2
	}
	if = {
		limit = {
			faith = {
				faith_hostility_level = {
					target = scope:councillor.faith
					value >= faith_evil_level
				}
			}
		}
		add = 3
	}
}

		}

	}

magic_focused_vassal_magi_grant_bonus = {
	if = {
		limit = {
			any_in_list = {
				list = titles_to_grant
				tier >= tier_county
			}
		}

		every_in_list = {
			list = titles_to_grant
			limit = {
				tier = tier_county
			}
			add = 5
		}

		every_in_list = {
			list = titles_to_grant
			limit = {
				tier = tier_duchy
			}
			add = 20
		}

		every_in_list = {
			list = titles_to_grant
			limit = {
				tier > tier_duchy
			}
			add = 50
		}
	}
}
