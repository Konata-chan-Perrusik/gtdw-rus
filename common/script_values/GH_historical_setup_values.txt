
# Slow historical development growth; use this should generally be the 'default' value
historical_development_growth_slow_value = {
	if = {
		limit = { 
			title_province = {
				OR = {
					GH_any_farmlands_terrain_trigger = yes
					GH_any_floodplains_terrain_trigger = yes
					GH_any_oasis_terrain_trigger = yes
				}				
			} 
		}
		value = 4
	}
	else_if = {
		limit = { 
			title_province = {
				OR = {
					GH_any_desert_terrain_trigger = yes
					GH_any_mountains_terrain_trigger = yes
					GH_any_jungle_terrain_trigger = yes
				}				
			} 
		}
		value = 1
	}
	else_if = {
		limit = {
			title_province = {
				OR = {
					GH_any_hills_terrain_trigger = yes
					GH_any_wetlands_terrain_trigger = yes
					GH_any_steppe_terrain_trigger = yes
				}
			}
		}
		value = 1.5
	}
	else = {
		value = 2
	}
	if = {
		limit = { title_province = { has_holding_type = metro_holding } }
		if = {
			limit = { holder = { highest_held_title_tier = tier_empire } } # Imperial capitals should get a slightly larger boost
			multiply = 1.75
		}
		else = { multiply = 1.5 }
	}
}
