arcana_lifestyle_progress_pulse = {
	trigger = {
		has_lifestyle = arcana_lifestyle
		is_playable_character = yes # lifestyles inhernetly assume you have one
		exists = capital_province # Sometimes you can have a title but no province
		trigger_if = {
			limit = {
				is_ai = no
			}
			has_had_focus_for_days > 30
		}
		NOR  = {
			has_character_flag = had_recent_lifestyle_ongoing_event
			# GH_TODO add flag for arcana event chains
		}
	}

	first_valid_on_action = {
		arcana_lifestyle_ongoing_events # General events
		arcana_lifestyle_unavailable_events # Special events for unavilable characters
	}
}

arcana_lifestyle_ongoing_events = {
	trigger = {
		is_available = yes
	}

	random_on_action = {
		chance_to_happen = 10

		80 = arcana_lifestyle_common_events
		15 = arcana_lifestyle_uncommon_events
		5  = arcana_lifestyle_rare_events
	}
}

	}
}
