	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_realm.dds"
	}
	decision_group_type = sjalvolki
	desc = convert_realm_feudalism_rite_desc

	ai_check_interval = 36

	is_shown = {
		OR = {
			GH_is_any_tribal_government_trigger = yes
			GH_gov_is_tribal_clan_trigger = yes
		}
		OR = {
			faith.religion = religion:sjalvolki_religion
			culture = { has_cultural_pillar = heritage_sjalvolki }
		}
	}

	is_valid = {
		is_any_rite_faith_trigger = yes
		capital_province.county = {
			development_level >= convert_to_feudalism_development_requirement
		}
	}

	effect = {
		trigger_event = major_decisions.0001
		save_scope_as = actor
		add_prestige = 200
		add_piety = 200
		convert_whole_realm_to_feudalism_effect = yes
		GH_anewworld_check = yes
		hidden_effect = {
			every_vassal = {
				trigger_event = {
					id = major_decisions.0002
					days = 3
				}
			}
		}
	}

	ai_will_do = {
		base = 100
	}
}

		is_any_rite_faith_trigger = yes

			text = form_waymakers_of_velos_decision.tt.is_in_correct_region

				geographical_region = world_ga_aversarian_homelands
			}
		}
		NOT = {
			capital_province = {
				has_special_building = yes
			}
		}
	}

	is_valid_showing_failures_only = {
		is_available_adult = yes
	}

	effect = {
		save_scope_as = founder
		show_as_tooltip = { waymakers_of_velos_formed_decision_effect = yes }
		custom_tooltip = waymakers_of_velos_formed_tooltip
		trigger_event = { id = sjalvolki_decision.2001 }
		hidden_effect = {
			add_to_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:form_waymakers_of_velos_decision
			}
		}
		set_global_variable = {
			name = waymakers_title
			value = scope:founder.primary_title
		}
	}

		piety = 1000
		prestige = 1000
	}

	}

		base = 100
	}
