# BS - Announce Aversaria Voreias
# by ThePinkPanzer
announce_aversaria_voreias_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_aversaria.dds"
	}
	ai_check_interval = 36
	desc = announce_aversaria_voreias_decision_desc
	decision_group_type = story

	is_shown = {
		exists = capital_province
		culture = { has_cultural_pillar = heritage_aversarian }
		has_game_rule = story_content_enabled
		character_black_sun_trigger = yes
		current_date >= 1254.1.1	
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:announce_aversaria_voreias_decision
			}
		}
	}

	is_valid = {
		culture = { has_cultural_pillar = heritage_aversarian }
		has_title = title:k_niidinia
		OR = {
			AND = {
				has_title = title:k_vyshtanka
				has_title = title:k_heartwood
			}
			AND = {
				has_title = title:k_rohr
				has_title = title:k_marroux
			}
		}
	}

	effect = {
		save_scope_as = announce_aversaria_voreias
		
		custom_tooltip = announce_aversaria_voreias_decision_tt
		hidden_effect = {
			get_title = title:e_aversaria_voreias
			title:e_aversaria_voreias = {
				copy_title_history = title:k_epavarion
			}
			set_primary_title_to = title:e_aversaria_voreias
			destroy_title = title:k_epavarion
			title:e_gurdurbok = {
				every_in_de_jure_hierarchy = {
					limit = { tier = tier_kingdom }
					set_de_jure_liege_title = title:e_aversaria_voreias
				}
			}
			destroy_title = title:e_gurdurbok
			trigger_event = black_sun.0001
		}
	
		if = {
			limit = { character_black_sun_trigger = yes }
			add_character_flag = formed_aversaria_voreias
		}

		if = {
			limit = { NOT = { has_dlc_feature = legends } }
			dynasty = {
				add_dynasty_perk = black_sun_legacy_1
			}
		}

		every_player = {
			limit = {
				NOT = { this = scope:announce_aversaria_voreias }
				in_diplomatic_range = scope:announce_aversaria_voreias
			}
		}

		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:announce_aversaria_voreias_decision
		}
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
