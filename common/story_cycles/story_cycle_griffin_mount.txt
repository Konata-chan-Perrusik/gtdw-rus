story_cycle_griffon_mount = {
	
	on_setup = {
		add_to_variable_list = {
			name = familiar_artifacts
			target = story_owner.var:familiar_artifact
		}
	}

	on_end = {
		every_in_list = {
			variable = familiar_artifacts
			destroy_artifact = scope:griffon_artifact
		}
	}

	on_owner_death = {
		scope:story = { end_story = yes }
	}

	effect_group = {
		days = { 15000 16000 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = scope:griffon
				scope:story_owner = {
					any_character_artifact = {
						this = scope:griffon_artifact
					}
				}
			}
			effect = {
				story_owner = {
					trigger_event = marcher.0101
				}
			}
		}
	}
}
