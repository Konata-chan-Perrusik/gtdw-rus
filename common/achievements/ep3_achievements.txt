# 02. Band of Brothers
ep3_02_band_of_brothers_achievement = {
	possible = {
		
	}
	happened = {
		custom_description = {
			text = band_of_brothers_trigger
			is_landless_adventurer = yes
			any_courtier = {
				count >= 10
				is_alive = yes
			}
			any_courtier = {
				count = all
				opinion = {
					target = root
					value >= 40 # sync with define:NDomicile|TEMPERAMENT_THRESHOLD_HIGH
				}
			}
		}
	}
}
